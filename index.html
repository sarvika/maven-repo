Add the repository to your project 

            
<repositories>
    <repository>
        <id>sarvika-maven-repo</id>
        <url>https://rawgit.com/sarvika/maven-repo/master</url>
        <releases>
            <enabled>true</enabled>
        </releases>
        <snapshots>
            <enabled>true</enabled>
        </snapshots>
    </repository>
</repositories>
